    <div id="ui">
        <div>Skor: <span id="score">0</span></div>
        <div>Can: <span id="lives">3</span></div>
    </div>

    <div id="gameOverScreen">
        <h2>OYUN BİTTİ</h2>
        <p>Toplam Skorun: <span id="finalScore">0</span></p>
        <button id="restartBtn" onclick="resetGame()">Yeniden Başla</button>
    </div>

    <canvas id="gameCanvas" width="400" height="500"></canvas>

    <script>
        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");

        // Oyun Değişkenleri
        let score = 0;
        let lives = 3;
        let gameActive = true;

        // Sepet Özellikleri
        const basket = {
            x: 175,
            y: 450,
            width: 70,
            height: 20,
            speed: 8
        };

        // Düşen Altın Özellikleri
        let gold = {
            x: Math.random() * (canvas.width - 20),
            y: 0,
            size: 15,
            speed: 4
        };

        // Klavye Kontrolleri
        let keys = { Right: false, Left: false };

        document.addEventListener("keydown", (e) => {
            if (e.key === "ArrowRight" || e.key === "d") keys.Right = true;
            if (e.key === "ArrowLeft" || e.key === "a") keys.Left = true;
        });

        document.addEventListener("keyup", (e) => {
            if (e.key === "ArrowRight" || e.key === "d") keys.Right = false;
            if (e.key === "ArrowLeft" || e.key === "a") keys.Left = false;
        });

        // Oyunu Sıfırlama
        function resetGame() {
            score = 0;
            lives = 3;
            gameActive = true;
            gold.y = 0;
            gold.x = Math.random() * (canvas.width - 20);
            gold.speed = 4;
            document.getElementById("gameOverScreen").style.display = "none";
            document.getElementById("score").innerText = score;
            document.getElementById("lives").innerText = lives;
            update();
        }

        // Ana Oyun Döngüsü
        function update() {
            if (!gameActive) return;

            // 1. Sepeti Hareket Ettir
            if (keys.Right && basket.x < canvas.width - basket.width) {
                basket.x += basket.speed;
            }
            if (keys.Left && basket.x > 0) {
                basket.x -= basket.speed;
            }

            // 2. Altını Aşağı Düşür
            gold.y += gold.speed;

            // 3. Yakalama Kontrolü (Çarpışma)
            if (gold.y + gold.size >= basket.y && 
                gold.x + gold.size >= basket.x && 
                gold.x <= basket.x + basket.width) {
                
                score += 10;
                document.getElementById("score").innerText = score;
                gold.y = 0;
                gold.x = Math.random() * (canvas.width - 20);
                
                // Oyun gittikçe hızlansın
                if (score % 50 === 0) gold.speed += 0.5; 
            }

            // 4. Kaçırma Kontrolü
            if (gold.y > canvas.height) {
                lives--;
                document.getElementById("lives").innerText = lives;
                gold.y = 0;
                gold.x = Math.random() * (canvas.width - 20);

                if (lives <= 0) {
                    gameActive = false;
                    document.getElementById("finalScore").innerText = score;
                    document.getElementById("gameOverScreen").style.display = "flex";
                }
            }

            // Çizim İşlemleri
            ctx.clearRect(0, 0, canvas.width, canvas.height); // Ekranı temizle

            // Sepeti Çiz
            ctx.fillStyle = "#8B4513";
            ctx.fillRect(basket.x, basket.y, basket.width, basket.height);
            ctx.strokeStyle = "#fff";
            ctx.strokeRect(basket.x, basket.y, basket.width, basket.height);

            // Altını Çiz
            ctx.beginPath();
            ctx.arc(gold.x, gold.y, gold.size, 0, Math.PI * 2);
            ctx.fillStyle = "#FFD700";
            ctx.fill();
            ctx.strokeStyle = "#ffa502";
            ctx.stroke();
            ctx.closePath();

            requestAnimationFrame(update);
        }

        // Başlat
        update();
    </script>
